93  size_t knn_search(const Eigen::Vector4d& pt, int k, size_t* k_index, double* k_sq_dist) const {
94  if (points.empty()) {
95  return 0;
96  }
97 
98  std::priority_queue<std::pair<size_t, double>> queue;
99  for (size_t i = 0; i < points.size(); i++) {
100  const double sq_dist = (points[i] - pt).squaredNorm();
101  queue.push({i, sq_dist});
102  if (queue.size() > k) {
103  queue.pop();
104  }
105  }
106 
107  const size_t n = queue.size();
108  while (!queue.empty()) {
109  k_index[queue.size() - 1] = queue.top().first;
110  k_sq_dist[queue.size() - 1] = queue.top().second;
111  queue.pop();
112  }
113 
114  return n;
115  }
